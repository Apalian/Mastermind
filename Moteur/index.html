        def compare(essai, resultat):
        Compare = essai
        index = []
        for i in range (len(Compare)):
            if Compare[i] == resultat[i]:
                print("Un numéro bien placé :{}".format(essai[i]))
                index.append(i)
        for l in range (len(index)):
            Compare.pop(index[l])
        for i in range (len(resultat)):
            for n in range (len(Compare)):
                if Compare[n] == resultat[i]:
                    print("Un numéro bon :{}".format(Compare[n]))
